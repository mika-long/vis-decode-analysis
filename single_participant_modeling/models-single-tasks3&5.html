# origin is top left  
data_to_pixel_y <- function(data_y) {
  return(-395 * data_y + 410)
}
data_to_pixel_x <- function(data_x) {
  return (53.5 * data_x + 317.5)
}

# origin is bottom left 
pixel_to_phy_x <- function(pixel, pxMM){
  (pixel - 50) / pxMM
}
pixel_to_phy_y <- function(pixel, pxMM){
  (410 - pixel) / pxMM
}

# return visual angle in degrees and not radian
vis_angle <- function(size, distance){
  return(2 * atan(size / (2 * distance)) * 180 / pi)
}

# tolerance for numerical precision

Model Fitting Overview

  • Models can be fitted in visual angle space, as well as physical space
  • Within each space, the effect of “distance to the other axis” could be multiplicative or additive. This is thoroughly investigated using task 3. The model performing model is one where we assume error ~ N(mu, sigma) where mu is non-zero (i.e., biased), and log(sigma) = beta + d_i or in other words, sigma = exp(beta) * d_i

Task 5.X

\[\begin{aligned} \texttt{error_va}_i &\sim \mathcal{N}(0, \sigma_i) \\ \sigma_i &= f(d_i) \,\, \text{where} \,\, d_i = \texttt{va.ans.y}_i \end{aligned}\]

Note that since the default in brms is a log link function, i.e., \(\log(\sigma) = f(d_i) \to \sigma = \exp(f(d_i))\)

Option 1: \(\sigma_i = \exp(f(d_i)) = \exp(\alpha + \texttt{va.ans.y}_i) = \exp(\alpha) \times \exp(\log(\texttt{va.ans.y}_i)) = \exp(\alpha) \times \texttt{va.ans.y}_i\)

f <- bf(error_va ~ 0, 
        sigma ~ 1 + offset(log.va.ans.y)) 

p <- c(
  prior(normal(0, 1), class = Intercept, dpar = sigma)
)

Fit model:

model5.x.no.bias <- task5_df %>% mutate(error_va = va.select.x - va.ans.x, 
                                 log.va.ans.y = log(va.ans.y)) %>%
  # account for numerical precision 
  mutate(error_va = ifelse(abs(error_va) < tolerance, 0, error_va)) %>% 
  # account for bugs in task 5 
  filter(va.ans.y != 0) %>% 
  brm(
    formula = f,
    prior = p,
    data = .,
    chains = 4, 
    file = "models/task5.x.no.bias", 
    file_refit = "on_change", 
    save_pars = save_pars(all = TRUE)
)

Option 1 — no bias + sigma_i = beta * d_i

Math model:

\[\begin{align*} \text{error_va}_{i} &\sim \mathcal{N}(0, \sigma) \\ \sigma &= f(d_i) = \beta \cdot d_i \,\, \text{where} \,\, d_i = \texttt{va.ans.x}_i \end{align*}\]

f <- bf(error_va ~ 0,
        sigma ~ 1 + offset(log.va.ans.x))

p <- c(
  prior(normal(0, 1), class = Intercept, dpar = sigma) 
)

Fit the model:

model.3.no.bias.no.alpha <- task3_df %>%
